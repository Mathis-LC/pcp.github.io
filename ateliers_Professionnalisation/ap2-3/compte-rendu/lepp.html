Pour mettre en place le firewall

whereis ufw

Si non trouvé

sudo apt update

sudo apt install ufw

Configurer les règles du firewall

sudo ufw enable

sudo ufw app list

sudo ufw allow "Nginx Full"

sudo ufw allow OpenSSH

sudo ufw status

sudo nano /etc/nginx/sites-available/applifrais

server {
                    listen 80;
                    server_name 100.115.28.237;

                    root /var/www/applifrais/www;
                    index index.html;

                    access_log /var/log/nginx/applifrais_access.log;
                    error_log /var/log/nginx/applifrais_error.log;

                    # Retourne une erreur 404 si aucune ressource ne correspond au chemin
                    location / {
                        try_files $uri $uri/ =404;
                    }
                    location ~ \.php$ {
                        include snippets/fastcgi-php.conf;
                        fastcgi_pass unix:/run/php/php8.3-fpm.sock;
                    }
                    # Fichiers statiques (pas de réécriture)
                    location /public/ {
                        try_files $uri =404;
                    }

                    # Redirection du reste vers index.php
                    location / {
                        try_files $uri $uri/ /index.php?$query_string;
                    }
                }

sudo systemctl reload nginx.service
